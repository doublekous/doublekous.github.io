def register(request):
if request.method == ‘GET’:
return render(request, ‘register.html’)

if request.method == 'POST':
    data = request.POST
    form = UserRegisterForm(data)
    if form.is_valid():
        User.objects.create_user(username=form.cleaned_data.get
        ('username'), password=form.cleaned_data.get('password'))
    else:
        errors = form.errors
        return render(request, 'register.html', {'errors':errors})

from django.contrib.auth.models import User
from django import forms

class UserRegisterForm(forms.Form):
username = forms.CharField(max_length=10, min_length=2, required=True, error_message={‘required’: ‘姓名注册必填’})
password = forms.CharField(max_length=30, required=True, error_message={‘required’: ‘注册必填’})
passwoed2 = forms.CharField(max_length=30, required=True,
error_message={‘required’: 密码不一致})
def clean(self):
name = self.cleaned_data.get(‘username’)
user = User.objects.filter(username=name).first()
if user:
raise forms.ValidationError({‘username’: ‘该账号已经注册，请去登录’})
if self.cleaned_data[‘password’] != self.cleaned_data[‘password2]:
raise forms.ValidationError({‘password’: ‘密码不一致’})
return self.cleaned_data

M有SQL进阶练习

Posted on 2018-10-17

– 查询薪资最高的员工姓名和工资
select ename,sal from TbEmp
where sal=(select max(sal) from TbEmp);

– 查询员工的姓名和年薪((月薪+补贴)12)
select ename as 姓名,(sal+if(comm,comm,0)) 12 as 年薪 from TbEmp;

– 查询有员工的部门的编号和人数
select dno as 部门编号, count(eno) as 人数 from TbEmp group by dno;

– 查询所有部门的名称和人数

select dname,ifnull(total,0)
from TbDept t1 left join
(select dno,count(dno) as total from TbEmp group by dno) t2
on t1.dno=t2.dno;
– 查询薪资最高的员工(Boss除外)的姓名和工资
select ename,sal from TbEmp
where sal=(select max(sal) from TbEmp where mgr is not null);

– 查询薪水超过平均薪水的员工的姓名和工资
select ename,sal from TbEmp
where sal>(select avg(sal) from TbEmp);

– 查询薪水超过其所在部门平均薪水的员工的姓名、部门编号和工资
select ename,t1.dno,sal,avgsal from TbEmp t1,
(select dno,avg(sal) as avgsal from TbEmp group by dno) t2
where t1.dno=t2.dno and sal>avgsal;

select ename,dname,sal from TbEmp t1 inner join
(select dno,avg(sal) as avgsal from TbEmp group by dno) t2
on t1.dno=t2.dno and sal>avgsal inner join TbDept t3
on t3.dno=t2.dno;

– 查询部门中薪水最高的人姓名、工资和所在部门名称

– 查询主管的姓名
select ename,job from TbEmp
where eno in (select distinct mgr from TbEmp where mgr is not null);

select ename,job from TbEmp t1
where exists(select ‘x’ from TbEmp t2 where t1.eno=t2.mgr);

– 说明：去重操作和集合效率是非常低下的
– 通常建议用exists或者not exists操作来替代去重和集合运算。
