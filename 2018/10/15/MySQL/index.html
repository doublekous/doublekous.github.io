#数据库 - database - 数据的仓库（集散地）

  • 通过数据库可以实现数据的持久化

  • 文件操作（读写文件）- 文本文件 /

    理论基础：关系代数和集合论
    具体表象：用二维表来组织数据
    学生表
    行：记录（1001 吕耐芯 1996年1月 男 四川乐山）
    列：字段（学号，姓名，出生日期，性别，家庭住址）
    有自己的编程语言 - SQL(结构化查询语言)

    • DDL (数据定义语言)：create（创建） / drop(删除) / alter（修改）
    • DML（数据操作语言）：insert(插入) / delete(删除) /update(查找) /select(收集)
    • DCL（数据控制语言）：grant（） /revoke（召回）

create database school default charset utf8; 创建学校数据库并且设置编码为utf8
drop database if exists school; 如果存在就删除
use school;
create table tb_student #创建二维表保存数据
( #列名 数据类型 约束条件 default（默认值约束）
stuid int not null,
stuname varchar(20) not null,
gender bit not null default 1,
birth date,
addr varchar(50) not null,
primary key (stuid) #主键 - 表中能够唯一标识一条记录的列
); # 主键约束 - primary key

修改表
alter table tb_student add column tel char(11) not null;
alter table tb_student drop column tel;
插入数据（单行插入）
insert into tb_student values
(10001, ‘吕耐芯’, 1, ‘19960103’, ‘四川乐山’)
插入数据（多行插入）
insert into tb_student(‘可以填上对应的值’) values
(10001, ‘吕耐芯’, 1, ‘19960103’, ‘四川乐山’),
(10002, ‘吕耐芯’, 1, ‘19960103’, ‘四川乐山’),
(10003, ‘吕耐芯’, 1, ‘19960103’, ‘四川乐山’);
删除数据
delete from tb_student where stuid=10001;
delete from tb_student where stuname=’吕耐芯帅’
删除区间数据
delete from tb_student where stuid in (10001, 10002);
截断数据（删除全表）
truncate table tb_student;
更新数据
update tb_student set addr=’四川绵阳’, gender=0 where stuid in (10006, 10007);

查询所有
select * from tb_student;
查询stuname和gender
select stuname, gender from tb_student;
